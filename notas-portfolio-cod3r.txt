Aula 11 - Tabela de Tecnologias

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
  directUrl = env("DIRECT_URL")
}

model Tecnologia {
  id        Int @id @default(autoincrement())
  nome      String
  descricao String
  imagem    String
  destaque  Boolean @default(false)

  @@map("tecnologias")
}
